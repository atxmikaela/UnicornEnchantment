import { useEffect, useState } from "react";
import { useDispatch, useSelector } from "react-redux"
import { useParams } from "react-router-dom";
import { receiveCornholeAction, receiveCornholeThunk } from "../../../store/cornholes";




const SingleCornhole = () => {


        const dispatch = useDispatch(receiveCornholeAction);
        const [isLoaded, setIsLoaded] = useState(false);
        const { id } = useParams();



        const cornhole = useSelector((state) => state.receiveCornholeAction)




        useEffect(() => {
            const receiveCornhole = async () => {
              await dispatch(receiveCornholeThunk(cornhole));
              setIsLoaded(true);
            };

            if (!isLoaded) {
              receiveCornhole();
            }
          }, [dispatch, isLoaded]);


    return (
        <>
        <h1>Everything you need to know about this special Cornhole</h1>
        <div>{cornhole.id}</div>
        <div>{id}</div>
        </>
    )





}


export default SingleCornhole;
